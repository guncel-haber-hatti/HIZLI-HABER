// backend/server.js const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const Parser = require('rss-parser'); const News = require('./models/News'); const http = require('http'); const { Server } = require("socket.io"); require('dotenv').config(); const app = express(); app.use(cors()); app.use(express.json()); mongoose.connect(process.env.MONGO_URI, { useNewUrlParser: true, useUnifiedTopology: true }); const server = http.createServer(app); const io = new Server(server, { cors: { origin: "*" } }); io.on("connection", (socket) => console.log("Bir kullanıcı bağlandı")); // Haber model // backend/models/News.js const newsSchema = new mongoose.Schema({ title: String, summary: String, content: String, image: String, language: { type: String, default: 'tr' }, views: { type: Number, default: 0 }, createdAt: { type: Date, default: Date.now } }); const NewsModel = mongoose.model('News', newsSchema); // RSS ile haber çekme async function fetchNews() { const parser = new Parser(); const feeds = [ 'https://www.hurriyet.com.tr/rss/gundem', 'http://feeds.bbci.co.uk/news/world/rss.xml' ]; for (let url of feeds) { const feed = await parser.parseURL(url); for (let item of feed.items) { const exists = await NewsModel.findOne({ title: item.title }); if (!exists) { const news = new NewsModel({ title: item.title, summary: item.contentSnippet, content: item.content, image: item.enclosure?.url || '', language: url.includes('bbc') ? 'en' : 'tr' }); await news.save(); io.emit("new-news", news); } } } } setInterval(fetchNews, 1000 * 60 * 60); fetchNews(); // API app.get('/api/news', async (req, res) => { const news = await NewsModel.find().sort({ createdAt: -1 }).limit(50); res.json(news); }); app.post('/api/news/:id/view', async (req, res) => { const news = await NewsModel.findById(req.params.id); if (news) { news.views = (news.views || 0) + 1; await news.save(); } res.sendStatus(200); }); server.listen(process.env.PORT || 5000, () => console.log('Sunucu çalışıyor'));
