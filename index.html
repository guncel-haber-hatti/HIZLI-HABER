// mega-news.js const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const Parser = require('rss-parser'); const http = require('http'); const { Server } = require('socket.io'); const app = express(); app.use(cors()); app.use(express.json()); const mongoURI = process.env.MONGO_URI || 'mongodb://127.0.0.1:27017/megaNews'; mongoose.connect(mongoURI).then(()=>console.log('MongoDB bağlı')).catch(err=>console.log(err)); const server = http.createServer(app); const io = new Server(server, { cors: { origin: "*" } }); const newsSchema = new mongoose.Schema({ title: String, summary: String, content: String, image: String, language: { type: String, default: 'tr' }, views: { type: Number, default: 0 }, createdAt: { type: Date, default: Date.now } }); const News = mongoose.model('News', newsSchema); // Özel Engin Savcu Haberini Sisteme Ekleme Fonksiyonu async function checkSpecialNews() { const specialTitle = "Engin Savcu'dan Kamuoyuna Samimi Açıklama"; const exists = await News.findOne({ title: specialTitle }); if (!exists) { const specialNews = new News({ title: specialTitle, summary: "Geçtiğimiz günlerde yaşanan olayla ilgili derin bir üzüntü içerisindeyim.", content: "Geçtiğimiz günlerde yaşanan ve maksadını aşan talihsiz olayla ilgili olarak derin bir üzüntü içerisindeyim. İstemeden de olsa kırdığım ve üzdüğüm tüm taraflardan içtenlikle özür dilerim. Bu süreç benim için önemli bir ders oldu.", image: "https://via.placeholder.com/800x400.png?text=Engin+Savcu+Aciklama", // Buraya gerçek bir görsel URL'si koyabilirsin language: 'tr' }); await specialNews.save(); console.log("Özel haber veritabanına eklendi."); } } async function fetchNews(){ await checkSpecialNews(); // Önce bizim özel haberi kontrol et const parser = new Parser(); const feeds = ['https://www.hurriyet.com.tr/rss/gundem', 'http://feeds.bbci.co.uk/news/world/rss.xml']; for(let url of feeds){ try { const feed = await parser.parseURL(url); for(let item of feed.items){ const exists = await News.findOne({ title: item.title }); if(!exists){ const news = new News({ title: item.title, summary: item.contentSnippet, content: item.content, image: item.enclosure?.url || '', language: url.includes('bbc') ? 'en' : 'tr' }); await news.save(); io.emit('new-news', news); } } } catch(e) { console.log("Haber çekme hatası:", url); } } } setInterval(fetchNews, 1000*60*60); fetchNews(); app.get('/news', async(req,res)=>res.json(await News.find().sort({createdAt:-1}).limit(50))); app.post('/news/:id/view', async(req,res)=>{ const n = await News.findById(req.params.id); if(n){ n.views=(n.views||0)+1; await n.save(); } res.sendStatus(200); }); app.get('/', (req,res)=>{ res.send(` Mega Haber

MEGA HABER

`); }); const PORT = process.env.PORT || 5000; server.listen(PORT, ()=>console.log('Mega Haber Sistemi aktif:', PORT));
