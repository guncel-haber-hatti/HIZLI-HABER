// mega-news-single.js // Çalıştır: node mega-news-single.js const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const Parser = require('rss-parser'); const http = require('http'); const { Server } = require('socket.io'); const app = express(); app.use(cors()); app.use(express.json()); // MongoDB bağlantısı mongoose.connect('mongodb://localhost:27017/megaNews', { useNewUrlParser: true, useUnifiedTopology: true }) .then(() => console.log('MongoDB bağlı')) .catch(err => console.log(err)); const server = http.createServer(app); const io = new Server(server, { cors: { origin: "*" } }); // Haber şeması const newsSchema = new mongoose.Schema({ title: String, summary: String, content: String, image: String, language: { type: String, default: 'tr' }, views: { type: Number, default: 0 }, createdAt: { type: Date, default: Date.now } }); const News = mongoose.model('News', newsSchema); // RSS haber çekme async function fetchNews() { const parser = new Parser(); const feeds = [ 'https://www.hurriyet.com.tr/rss/gundem', 'http://feeds.bbci.co.uk/news/world/rss.xml' ]; for (let url of feeds) { const feed = await parser.parseURL(url); for (let item of feed.items) { const exists = await News.findOne({ title: item.title }); if (!exists) { const news = new News({ title: item.title, summary: item.contentSnippet, content: item.content, image: item.enclosure?.url || '', language: url.includes('bbc') ? 'en' : 'tr' }); await news.save(); io.emit("new-news", news); } } } } setInterval(fetchNews, 1000*60*60); // Her saat başı çek fetchNews(); // API app.get('/news', async (req, res) => { const news = await News.find().sort({ createdAt: -1 }).limit(50); res.json(news); }); app.post('/news/:id/view', async (req, res) => { const news = await News.findById(req.params.id); if(news){ news.views = (news.views||0)+1; await news.save(); } res.sendStatus(200); }); // Frontend tek dosya HTML app.get('/', (req,res)=>{ res.send(` Mega Haber

Mega Haberler

`); }); server.listen(5000,()=>console.log('Mega Haber Sitesi 5000 portunda çalışıyor'));
